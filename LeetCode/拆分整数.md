```
def expanded_form(num):
    ls = str(num)
    out_string = ''
    list_num = []
    for i in ls:
        list_num.append(i)
    for i in range(len(list_num)):
        if list_num[i] != '0':
            out_string = out_string + list_num[i] + '0'*(len(list_num)-i-1) + ' + '
    return out_string[:-3]
```

```
def expanded_form(num):
    num = list(str(num))
    return ' + '.join(x + '0' * (len(num) - y - 1) for y,x in enumerate(num) if x != '0')
```
